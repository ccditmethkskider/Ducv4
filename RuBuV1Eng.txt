-- Tải danh sách key từ GitHub
local keyListUrl = "https://raw.githubusercontent.com/lnaa323sda/scki343/refs/heads/main/Key"
local validKeys = {}

-- Đọc danh sách key từ GitHub
local success, response = pcall(function()
    return game:HttpGet(keyListUrl)
end)

if not success then
    warn("Không thể tải danh sách key từ GitHub")
    return
end

-- Chuyển chuỗi key thành bảng
for key in string.gmatch(response, "[%w%-]+") do
    validKeys[key] = true
end

-- Kiểm tra key
if not validKeys[getgenv().key] then
    game.Players.LocalPlayer:Kick("Sai key")
    return
end

print("Chào mừng bạn! Key hợp lệ!")
